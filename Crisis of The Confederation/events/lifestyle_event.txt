	capable_only = yes
	
	trigger = {
		trait = athlete
		trait = maimed
	}
	
	mean_time_to_happen = {
		months = 120
	}
	
	# Dwell on the old glory days
	option = {
		name = "EVTOPTALIFESTYLE1700"
		add_trait = depressed
		ai_chance = { factor = 0 } # Never take this one if a better option is available.
	}
	
	# Give up on athletics, adopt a new hobby
	option = {
		trigger = {
			NOT = { trait = neurotic }
			NOT = { trait = stressed }
			NOT = { trait = close_minded }
			NOT = { trait = proud }
		}
		name = "EVTOPTBLIFESTYLE1700"
		remove_trait = athlete
	}
	
	# Buy a cybernetic replacement
	option = {
		trigger = {
			wealth = 500
			OR = {
				NOT = { religion_group = religious_group }
				trait = pragmatic
				piety = 100
			}
			NOT = {
				AND = {
					religion_group = religious_group
					trait = radical
				}
			}
		}
		name = "EVTOPTCLIFESTYLE1700"

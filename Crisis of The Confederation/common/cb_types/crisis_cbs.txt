# Generic Civil War CB
civil_war = {
	# A generic civil war triggered by a failed coup or purge
	name = CB_NAME_CIVIL_WAR
	war_name = WAR_NAME_CIVIL_WAR
	sprite = 3
	truce_days = 0
	hostile_against_others = no
	is_permanent = yes
	is_revolt_cb = yes
	can_ask_to_join_war = yes
	can_call_vassals = yes
	
	press_claim = yes
	
	can_use = {
		# Triggered by event only
		OR = {
			ROOT = { has_character_flag = flag_launching_civil_war }
			FROM = { has_character_flag = flag_launching_civil_war }
		}
	}
	
	can_use_title = {
		OR = {
			is_primary_holder_title = yes
			ROOT = { plot_target_title = { title = PREVPREV } }
		}
	}
	
	on_add = {
		log = "[Root.GetBestName] declared war on [From.GetBestName] using the CB civil_war." 
		# Remove initiating flags
		ROOT = { clr_character_flag = flag_launching_civil_war }
		FROM = { clr_character_flag = flag_launching_civil_war }
		
		# If FROM and ROOT share the same ideology, change that ideology's moral authority.
		if = {
			limit = {
				FROM = { primary_title = { tier = EMPEROR } }
				FROM = { religion = ROOT }
			}
			FROM = { religion_authority = { modifier = allowed_civil_war_empire } }
		}
		if = {
			limit = {
				FROM = { primary_title = { tier = KING } }
				FROM = { religion = ROOT }
			}
			FROM = { religion_authority = { modifier = allowed_civil_war_kingdom } }
		}
		
		# Downgrade FROM's public order
		FROM = {
			any_demesne_title = { limit = { has_law = public_order_1 } }
			add_law = public_order_0
		}
	}
	
	on_success = {
		log = "[Root.GetBestName] won the civil_war war against [From.GetBestName]."
		# The rebels have won!
		any_attacker = {
			limit = { character = ROOT }
			participation_scaled_prestige = 200
		}
		any_attacker = {
			limit = { NOT = { character = ROOT } }
			hidden_tooltip = { participation_scaled_prestige = 200 }
		}
		
		# Determine incoming government type
		if = {
			limit = {
				# These should be preserved
				ROOT = {
					NOT = { government = merchant_republic_government }
					NOT = { government = pirate_kingdom }
					NOT = { government = pirate_republic }
					NOT = { government = hivemind_government }
					NOT = { government = colonial_government_government }
				}
			}
			# Otherwise, the incoming government should use ROOT's power base and FROM's legitimacy type
			if = {
				limit = { FROM = { is_hereditary_trigger = yes } }
				if = {
					limit = { ROOT = { is_military_trigger = yes } }
					ROOT = { set_government_type = military_monarchy }
				}
				if = {
					limit = { ROOT = { is_corporate_trigger = yes } }
					ROOT = { set_government_type = corporate_monarchy }
				}
				if = {
					limit = { ROOT = { is_administrative_trigger = yes } }
					ROOT = { set_government_type = administrative_monarchy }
				}
			}
			if = {
				limit = { FROM = { is_elective_trigger = yes } }
				if = {
					limit = { ROOT = { is_military_trigger = yes } }
					ROOT = { set_government_type = military_republic }
				}
				if = {
					limit = { ROOT = { is_corporate_trigger = yes } }
					ROOT = { set_government_type = corporate_republic }
				}
				if = {
					limit = { ROOT = { is_administrative_trigger = yes } }
					ROOT = { set_government_type = administrative_republic }
				}
			}
			if = {
				limit = { FROM = { is_appointed_trigger = yes } }
				if = {
					limit = { ROOT = { is_military_trigger = yes } }
					ROOT = { set_government_type = military_directorate }
				}
				if = {
					limit = { ROOT = { is_corporate_trigger = yes } }
					ROOT = { set_government_type = corporate_directorate }
				}
				if = {
					limit = { ROOT = { is_administrative_trigger = yes } }
					ROOT = { set_government_type = administrative_directorate }
				}
			}
		}
		
		# Remove coup flags
		ROOT = { clr_character_flag = flag_failed_coup }
		FROM = { clr_character_flag = flag_failed_purge }
	}
	
	on_success_title = {
		usurp_title_plus_barony_if_unlanded = { target = ROOT type = revolt } # Messy solution, should be cleaned up later
		
		# New provisional government takes office.
		if = {
			limit = { ROOT = { is_merchant_republic = no } }
			add_law = succ_feudal_elective
			add_law = true_cognatic_succession
		}
		if = {
			limit = {
				crownlaw_title = { title = PREV }
				OR = {
					has_law = crown_term_length_0
					has_law = crown_term_length_1
					has_law = crown_term_length_2
					has_law = crown_term_length_3
				}
			}
			add_law = crown_term_length_4
		}
		if = {
			limit = {
				NOT = { crownlaw_title = { title = PREV } }
				OR = {
					has_law = local_term_length_0
					has_law = local_term_length_1
					has_law = local_term_length_2
					has_law = local_term_length_3
				}
			}
			add_law = local_term_length_4
		}
	}
	
	on_fail = {
		# Initiator fails to achieve whatever it was they were after
		ROOT = {
			cancel_plot = plot_coup
			cancel_plot = plot_justified_coup
		}
		
		# Remove coup flags
		ROOT = { clr_character_flag = flag_failed_coup }
		FROM = { clr_character_flag = flag_failed_purge }
	}
	
	on_reverse_demand = {
		# Ruler has won!
		FROM = { prestige = 100 }
		# Imprison all rebels
		any_attacker = { imprison = FROM }
		# Imprison all of ROOT's co-conspirators (if not rebels)
		if = {
			limit = {
				ROOT = {
					OR = {
						has_plot = plot_coup
						has_plot = plot_justified_coup
					}
					plot_target_char = { character = FROM }
				}
			}
			ROOT = {
				any_plot_backer = {
					limit = { NOT = { war_with = FROM } }
					imprison = FROM
				}
			}
		}
		# All anti-ruler vassals are automatically no longer anti-ruler
		FROM = {
			any_vassal = {
				limit = { trait = anti_ruler }
				hidden_tooltip = {
					remove_trait = anti_ruler
					character_event = { id = laws.5 }
				}
			}
			any_courtier = {
				limit = { trait = anti_ruler }
				hidden_tooltip = {
					remove_trait = anti_ruler
					character_event = { id = laws.5 }
				}
			}
		}
		# If FROM is in a constitutional crisis, this resolves it in their favor.
		if = {
			limit = { FROM = { has_character_modifier = constitutional_crisis } }
			FROM = { character_event = { id = laws.1000 tooltip = "EVTTOOLTIPLAWS1000" } }
		}
		# Coups fail immediately
		ROOT = {
			cancel_plot = plot_coup
			cancel_plot = plot_justified_coup
		}
		
		# If FROM and ROOTs' ideologies are different, the former gets a bonus and
		# the latter takes a hit
		
		# Remove coup flags
		ROOT = { clr_character_flag = flag_failed_coup }
		FROM = { clr_character_flag = flag_failed_purge }
	}
	
	on_attacker_leader_death = {
		if = {
			limit = { ROOT = { has_character_flag = flag_failed_coup } }
			end_war = invalid
		}
	}
	
	on_defender_leader_death = {
		if = {
			limit = { FROM = { has_character_flag = flag_failed_purge } }
			end_war = invalid
		}
	}

	attacker_ai_victory_worth = {
		factor = -1
	}
	
	attacker_ai_defeat_worth = {
		factor = 100
	}

	defender_ai_victory_worth = {
		factor = -1
	}
	
	defender_ai_defeat_worth = {
		factor = 100
	}
}

# Special CB for the crisis, so that the rebellious merchant republics can start off as such
crisis_of_the_confederation = {
	name = CB_NAME_INDEPENDENCE
	war_name = WAR_NAME_INDEPENDENCE
	sprite = 12
	truce_days = 3650
	
	is_permanent = yes
	is_independence = yes

	can_use = {
		always = no # Cannot use except when scripted
	}

	is_valid = {
		ROOT = {
			OR = {
				liege = { character = PREV } # either independent
				liege = { FROM = { is_liege_or_above = PREV } } # or have shared liege
			}
		}
		FROM = {
			any_demesne_title = {
				NOT = { lower_tier_than = ROOT }
				temporary = no
			}
			in_revolt = no
			liege_before_war = {
				in_revolt = no
				liege_before_war = { in_revolt = no }
			}
		}
	}
	
	on_add = {
		any_attacker = {
			any_demesne_title = {
				limit = { higher_tier_than = DUKE }
				FROM = {
					if = {
						limit = { supports_earth_trigger = yes }
						religion_authority = { modifier = allowed_major_rebellion years = 50 }
					}
				}
			}
		}
	}

	on_success = {
		log = "[Root.GetBestName] won the Crisis of the COnfederation against [From.GetBestName]."
		# Handle ideological effects
		any_attacker = {
			limit = {
				higher_tier_than = DUKE
				OR = {
					vassal_of = FROM
					liege_before_war = { character = FROM }
				}
			}
			if = {
				FROM = {
					if = {
						limit = {
							OR = {
								AND = {
									supports_earth_trigger = yes
									controls_earth_trigger = yes
								}
								supports_centralization_trigger = yes
							}
						}
						religion_authority = { modifier = fell_to_major_rebellion years = 60 }
						piety = -100
					}
					hidden_tooltip = {
						any_vassal = {
							limit = {
								OR = {
									AND = {
										supports_earth_trigger = yes
										FROM = { controls_earth_trigger = yes }
									}
									supports_centralization_trigger = yes
								}
							}
							opinion = { who = PREV modifier = opinion_lost_defensive_war }
						}
					}
				}
			}
			if = {
				limit = {
					OR = {
						AND = {
							supports_earth_trigger = yes
							FROM = { controls_earth_trigger = yes }
						}
						supports_centralization_trigger = yes
					}
					NOT = {
						AND = {
							opposes_earth_trigger = yes
							FROM = { controls_earth_trigger = yes }
						}
					}
					supports_secession_trigger = no
				}
				piety = -100
			}
			if = {
				limit = {
					OR = {
						AND = {
							opposes_earth_trigger = yes
							FROM = { controls_earth_trigger = yes }
						}
						supports_secession_trigger = yes
					}
				}
				religion_authority = { modifier = successful_major_rebellion years = 60 }
				piety = 100
			}
		}
		# Grant independence to the victorious rebels
		if = {
			limit = { FROM = { independent = yes } }
			any_attacker = {
				limit = {
					OR = {
						vassal_of = FROM
						liege_before_war = { character = FROM }
					}
				}
				set_defacto_liege = THIS
				hidden_tooltip = {
					prestige = 100
					any_demesne_title = {
						limit = {
							controls_religion = no
							higher_tier_than = BARON
						}
						add_weak_pressed_claim = FROM
					}
					set_character_flag = ai_flag_never_transfer
				}
			}
		}
		
		ROOT = {
			participation_scaled_prestige = 100
			hidden_tooltip = { disband_event_forces = faction_revolters }
		}
		any_attacker = {
			limit = { NOT = { character = ROOT } }
			hidden_tooltip = {
				participation_scaled_prestige = 100
				disband_event_forces = faction_revolters
			}
		}

		FROM = {
			hidden_tooltip = { disband_event_forces = faction_loyalists }
			prestige = -100
			if = {
				limit = { NOT = { has_dlc = "Zeus" } }
				any_demesne_title = { # All titles
					limit = {
						OR = {
							tier = king
							tier = emperor
						}
						NOT = { has_law = crown_authority_0 }
					}
					if = {
						limit = { has_law = crown_authority_1 }
						add_law = crown_authority_0
					}
					if = {
						limit = { has_law = crown_authority_2 }
						add_law = crown_authority_1
					}
					if = {
						limit = { has_law = crown_authority_3 }
						add_law = crown_authority_2
					}
					if = {
						limit = { has_law = crown_authority_4 }
						add_law = crown_authority_3
					}
				}
			}
		}
	}

	on_fail = {
		any_attacker = { set_defacto_liege = FROM }
		ROOT = {
			prestige = -50
#			prisoner = FROM
			add_character_modifier = { name = faction_independence_ultimatum_timer hidden = yes duration = 3650 }
			hidden_tooltip = { disband_event_forces = faction_revolters }
			participation_scaled_prestige = -50
		}
		any_attacker = {
			limit = { NOT = { character = ROOT } }
			hidden_tooltip = {
				participation_scaled_prestige = -50 
				disband_event_forces = faction_revolters
			}
			add_character_modifier = { name = faction_independence_ultimatum_timer hidden = yes duration = 3650 }
		}
		FROM = { hidden_tooltip = { disband_event_forces = faction_loyalists } }
	}

	on_reverse_demand = {
		any_attacker = { set_defacto_liege = FROM }
		# Handle ideological effects
		any_attacker = {
			limit = {
				higher_tier_than = DUKE
				OR = {
					vassal_of = FROM
					liege_before_war = { character = FROM }
				}
			}
			if = {
				FROM = {
					if = {
						limit = {
							OR = {
								AND = {
									supports_earth_trigger = yes
									controls_earth_trigger = yes
								}
								supports_centralization_trigger = yes
							}
						}
						religion_authority = { modifier = crushed_major_rebellion years = 60 }
						piety = 100
					}
					hidden_tooltip = {
						any_vassal = {
							limit = {
								OR = {
									AND = {
										supports_earth_trigger = yes
										FROM = { controls_earth_trigger = yes }
									}
									supports_centralization_trigger = yes
								}
							}
							opinion = { who = PREV modifier = opinion_won_defensive_war }
						}
					}
				}
			}
			if = {
				limit = {
					OR = {
						AND = {
							opposes_earth_trigger = yes
							FROM = { controls_earth_trigger = yes }
						}
						supports_secession_trigger = yes
					}
				}
				religion_authority = { modifier = failed_major_rebellion years = 60 }
				piety = -100
			}
		}
		ROOT = {
			prestige = -100
			prisoner = FROM
			add_character_modifier = { name = faction_independence_ultimatum_timer hidden = yes duration = 3650 }
			hidden_tooltip = { disband_event_forces = faction_revolters }
			participation_scaled_prestige = -100
		}
		
		any_attacker = {
			limit = { 
				NOT = { character = ROOT }
				liege = { character = FROM } 
			}
			prisoner = FROM
			participation_scaled_prestige = -100
			hidden_tooltip = {
				disband_event_forces = faction_revolters
				add_character_modifier = { name = faction_independence_ultimatum_timer hidden = yes duration = 3650 }
			}
		}
		
		FROM = {
			prestige = 50
			hidden_tooltip = { disband_event_forces = faction_loyalists }
		}
	}

	attacker_ai_victory_worth = {
		factor = -1 # always accept
	}
	
	attacker_ai_defeat_worth = {
		factor = 150
	}

	defender_ai_victory_worth = {
		factor = -1 # always accept
	}
	
	defender_ai_defeat_worth = {
		factor = 100
	}
}
